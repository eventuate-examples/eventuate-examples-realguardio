  participant "route.ts" as BFF

  participant "CustomerController" as CC
  participant "CustomerService" as CSVC
  database "Customer DB" as CDB

CSVC -> CDB : assignRole(\nCOMPANY_ROLE_ADMIN)
CSVC --> CC : CustomerAndCustomerEmployee
CC --> BFF : 200 OK
BFF --> C : CustomerAndCustomerEmployee
