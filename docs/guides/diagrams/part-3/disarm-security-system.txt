@startuml

participant "Client" as C

box "BFF (Next.js)"
  participant "route.ts" as BFF
end box

box "Security System Service"
  participant "SecuritySystemController" as SSC
  participant "SecuritySystemServiceImpl" as SSVC
  participant "LocalSecuritySystem\nActionAuthorizer" as AUTH
  participant "CustomerServiceClientImpl" as CSC
  database "Security System DB" as SSDB
end box

box "Customer Service"
  participant "LocationRoleController" as LRC
  participant "LocationRoleServiceImpl" as LRS
  database "Customer DB" as CDB
end box

C -> BFF : PUT /api/securitysystems/{id}\n{action: DISARM}
BFF -> SSC : PUT /securitysystems/{id}\nAuthorization: Bearer <JWT>
SSC -> SSVC : disarm(id)
SSVC -> SSDB : findById(id)
SSDB --> SSVC : SecuritySystem\n(locationId)

SSVC -> AUTH : isAllowed("disarm", id)
AUTH -> SSDB : findById(id)
SSDB --> AUTH : SecuritySystem\n(locationId)
AUTH -> CSC : getUserRolesAtLocation(\nuserId, locationId)
CSC -> LRC : GET /locations/{locationId}/roles\nAuthorization: Bearer <JWT>
LRC -> LRS : getUserRolesAtLocation(locationId)
LRS -> CDB : findDirectRoles +\nfindTeamRoles
CDB --> LRS : roles
LRS --> LRC : roles
LRC --> CSC : RolesResponse
CSC --> AUTH : Set<String> roles
AUTH -> AUTH : verify roles\ncontain required\npermission

SSVC -> SSVC : securitySystem.disarm()
SSVC -> SSDB : save(securitySystem)
SSVC --> SSC : SecuritySystem
SSC --> BFF : 200 OK
BFF --> C : SecuritySystem

@enduml
